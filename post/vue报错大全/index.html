###解决办法

1、根据Terminal报错提示安装@vitejs/plugin-vue

1

npm i @vitejs/plugin-vue

2、在项目根目录下新增vite.config.js文件，内容如下

1
2
3
4
5
6

import {defineConfig} from "vite";
import vue from "@vitejs/plugin-vue";

export default defineConfig({
    plugins: [vue()]
})
